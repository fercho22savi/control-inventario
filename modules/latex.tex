\section{Formato de Texto}
\begin{itemize}
    \item \texttt{\textbackslash textbf\{negrita\}} → \textbf{negrita}
    \item \texttt{\textbackslash textit\{cursiva\}} → \textit{cursiva}
    \item \texttt{\textbackslash underline\{subrayado\}} → \underline{subrayado}
\end{itemize}

\section{Listas}

\end{itemize}
\end{verbatim}

\subsection*{Numeradas}
\begin{verbatim}
\begin{enumerate}
  \item Elemento
\end{enumerate}
\end{verbatim}

\section{Matemáticas}
Fórmula en línea: \( a^2 + b^2 = c^2 \)

Fórmula en bloque:
\[
\int_0^1 x^2 dx = \frac{1}{3}
\]

\begin{itemize}
    \item \texttt{\textbackslash frac\{a\}\{b\}} → \( \frac{a}{b} \)
    \item \texttt{\textbackslash sqrt\{x\}} → \( \sqrt{x} \)
    \item \texttt{\textbackslash sum\_\{i=1\}\^\{n\}} → \( \sum_{i=1}^{n} \)
    \item \texttt{\textbackslash alpha, \textbackslash beta} → \( \alpha, \beta \)
\end{itemize}

\end{itemize}
